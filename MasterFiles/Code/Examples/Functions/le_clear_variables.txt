# le_clear_variables

a = 123
sysa = 234
le_system_variable(sysa,True)

assert(a,123)  # should pass
assert(sysa,234)  # should pass

le_clear_variables()
assert(a,123)  # should fail
assert(sysa,234)  # should pass

le_system_variable(sysa,False)
assert(sysa,234)  # should pass
le_clear_variables()
assert(sysa,234)  # should fail
